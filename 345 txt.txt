/*1. Write a program to accept any number up to six digits and print each digit in words.
For eg- 1503 = one five zero three
*/

void print_digit()
{
	int n,count=0,a[6],i;

	// Taking number input from user
	printf("enter the number");
	scanf("%d",&n);

	// Extracting digits one by one using modulus and storing them in array
	// Digits are stored in reverse order (last digit first)
	while(n){
		a[count]=n%10;
		n=n/10;
		count++;
	}

	// Printing digits in correct order by traversing array backwards
	for (i=count-1;i>=0;i--){

		// Converting each digit into word using switch case
		switch(a[i]){
		case 1:printf("One ");
			break;
		case 2:printf("Two ");
			break;
		case 3:printf("Three ");
			break;
		case 4:printf("Four ");
			break;
		case 5:printf("Five ");
			break;
		case 6:printf("Six ");
			break;
		case 7:printf("Seven ");
			break;
		case 8:printf("Eight ");
			break;
		case 9:printf("Nine ");
			break;
		case 0:printf("Zero ");
			break;
		}
	}
}

//Write a program to find out the day of week for a given data - 29 th Aug 1999

int day_week(){

	int d,m,y,total =0;

	// Array used for century correction values
	int a[4]={0,5,3,1};

	int x;

	// Month days arrays for leap year and non-leap year
	int lp[]={31,29,31,30,31,30,31,31,30,31,30,31};
	int np[]={31,28,31,30,31,30,31,31,30,31,30,31};

	// Taking date input
	printf("enter the date :");
	scanf("%d",&d);

	printf("enter the month :");
	scanf("%d",&m);

	printf("enter the year :");
	scanf("%d",&y);

	// Validating date depending on leap year or normal year
	if(y%100==0 &&y%400==0){

		// Leap year validation
		if(lp[m-1]>=d)
			printf("the date is valid\n");
		else{
			printf("the date is invalid\n");
			return 0;
		}
	}
	else{

		// Non-leap year validation
		if(np[m-1]>=d)
			printf("the entered date is valid\n");
		else{
			printf("the entered date is invalid\n");
			return 0;
		}
	}

	// Adding day contribution
	total =total +d%7;

	// Adding month code contribution
	switch(m){

	case 1:total =total+3;
			break;

	case 2:
		if(y%100==0 && y%400==0)
				total=total+4;
		else
				total=total+3;
	break;

	case 3:
		if(y%100==0 && y%400==0)
				total=total+7;
		else
			    total=total+6;
	break;

	// Similar month offsets for all months
	case 4: total += (y%100==0 && y%400==0)?9:8; break;
	case 5: total += (y%100==0 && y%400==0)?12:11; break;
	case 6: total += (y%100==0 && y%400==0)?14:13; break;
	case 7: total += (y%100==0 && y%400==0)?17:16; break;
	case 8: total += (y%100==0 && y%400==0)?20:19; break;
	case 9: total += (y%100==0 && y%400==0)?22:21; break;
	case 10: total += (y%100==0 && y%400==0)?25:24; break;
	case 11: total += (y%100==0 && y%400==0)?27:26; break;
	case 12: total += (y%100==0 && y%400==0)?30:29; break;
	}

	// Century correction calculation
	x=(y/100)%4;
	total=total+a[x];

	// Year contribution within century
	x=(y%100)-1;
	total=total+(x/4)*2+(x-x/4);

	// Final modulo to get weekday index
	total=total%7;

	// Printing weekday name
	switch(total){
	case 1:printf("the date specified is Monday\n"); break;
	case 2:printf("the date specified is Tuesday\n"); break;
	case 3:printf("the date specified is Wednesday\n"); break;
	case 4:printf("the date specified is Thursday\n"); break;
	case 5:printf("the date specified is Friday\n"); break;
	case 6:printf("the date specified is saturday\n"); break;
	case 0:printf("the date specified is sunday\n"); break;
	}

return 0;
}

//Write a program to find the difference of two dates in years, months and days?

int diff_dates(){

	// Variables for start date (d1,m1,y1) and end date (d2,m2,y2)
	int d1,m1,y1,d2,m2,y2;

	// Arrays storing number of days in each month for leap year and non-leap year
	int lp[]={31,29,31,30,31,30,31,31,30,31,30,31};
	int np[]={31,28,31,30,31,30,31,31,30,31,30,31};

	// Variables to store final difference in days, months and years
	int d=0,m=0,y=0;

	// Taking starting date input from user
	printf("enter the start date with spaces");
	scanf("%d %d %d",&d1,&m1,&y1);

	// Taking ending date input from user
	printf("enter the end date with spaces");
	scanf("%d %d %d",&d2,&m2,&y2);

	// Checking if the start year is leap year or not and validating start date
	if(y1%100==0 && y1%400==0){

		// Leap year: checking if day is within valid range of that month
		if(d1<1||d1>lp[m1-1]){
			printf("invalid date");
			return 0;
		}
	}
	else{

		// Non-leap year: checking if day is within valid range of that month
		if(d1<1||d1>lp[m1-1]){
			printf("invalid date");
			return 0;
		}
	}

	// Validating month values for both start and end dates
	if(m1>12 || m1<1 ||m2>12||m2<1)
	{
		printf("invalid date");
		return 0;
	}

	// If start year is greater than end year, swap the dates using XOR swapping
	if(y1>y2){
		d1^=d2^=d1^=d2;
		m1^=m2^=m1^=m2;
		y1^=y2^=y1^=y2;
	}

	// Calculating remaining days from start date to end of that month
	if(y1%100==0 && y1%400==0){
		d=d+lp[m1-1]-d1;
	}
	else
		d=d+np[m1-1]-d1;

	// Adding days passed in the ending month up to d2
	d=d+d2;

	// Moving start month forward and end month backward for month calculation
	m1=m1+1;
	m2=m2-1;

	// Calculating month difference between start and end dates
	m=m+12-m1;
	m=m+m2+1;

	// Adjusting years after excluding partial months
	y1++;
	y2--;

	// Calculating year difference
	y=y2-y1+1;

	// Printing final difference result
	printf("the difference in the given dates is %d days %d months %d years",d,m,y);

	return 0;
}

// Function to check PIN authentication
int checkpin(){

	int pin;

	// Asking user for ATM PIN before any operation
	printf("to perform the required operation enter the pin \n");
	scanf("%d",&pin);

	// If correct PIN entered, allow access
	if (pin == 1234)
		return 1;

	// If wrong PIN but attempts still remaining (max 3 tries)
	else if (cnt < 2){

		cnt ++;

		// Displaying remaining chances
		printf("you have entered the wrong pin %d more chances left before shutting the system \n",3-cnt);

		// Recursive call for retry
		checkpin();
	}

	// If wrong PIN entered 3 times, shutdown ATM
	else {

		printf("atm shutting due to wrong pin retry after sometime\n");

		// Resetting counter after shutdown
		cnt=0;

		return 0;
	}

return 1;
}

// Function to withdraw money
int withdrawal(int amount){

	int w;

	// Taking withdrawal amount input
	printf("enter the amount that you want to withdraw\n");
	scanf("%d",&w);

	// Checking if withdrawal amount exceeds balance
	if(amount<w){
		printf("insufficient balabnce");
		return amount;
	}

	// Deducting withdrawal amount
	amount =amount-w;

	printf("the amount %d is successfully withdrawn and the left balance is %d\n",w,amount);

	return amount ;
}

// Function to deposit money
int deposit(int amount)
{
	int d;

	// Taking deposit amount input
	printf("enter the amount that you want to deposit \n");
	scanf("%d",&d);

	// Adding deposit to account balance
	amount=amount+d;

	printf("the amount %d is successfully deposited in your account and the current balance is %d\n",d,amount);

	return amount;
}

/*1. Write program to print the following patterns . Pattern should be scalable, if input
	(number of lines) is 6 then the output should be.*/

	int l,i,j;

	// Taking number of lines from user
	printf("enter the no of lines the pattern should have");
	scanf("%d",&l);

	int n=65; // ASCII value of 'A'

	// Outer loop controls number of rows
	for(i=1;i<=l;i++){

		n=65; // Reset character to 'A' for each new row

		for(j=1;j<=i;j++){

			// Printing the current alphabet character
			printf("%c",n);

			if(j!=i)
				printf("*");

			n++; // Move to next alphabet character
		}

		// Moving to next line after each row
		printf("\n");
	}
}

void kp(){

	int n,a=65,i,j,p;

	// Taking number of lines input
	printf("enter no of lines");
	scanf("%d",&n);

	p=n; // Copy of n used for controlling second half

	// Printing upper half of pattern
	for (i=1;i<=n/2;i++){

		// Printing repeated alphabets with decreasing count
		for (j=i;j<=(n/2)+1;j++){
			printf("%c ",a);
		}

		printf("\n");

		a++; // Move to next character
		p--; // Reduce remaining lines counter
	}

	// Printing the middle single character line
	printf("%c\n",a);
	p--;

	// Printing lower half of the pattern (reverse)
	for (i=n/2;i>=1;i--){

		// Only continue if lines remain
		if (p!=0){

			a--; // Step back to previous character

			// Printing repeated alphabets for lower half
			for (j=i;j<=n/2+1;j++){
				printf("%c ",a);
			}

			p--;
			printf("\n");
		}
		else break;
	}
}

/*Write a program to calculate the sum and average of 10 positive numbers using continue for
skipping invalid input.*/
void sum_and_average()
{
	int n,x,s=0;

	// Taking number of inputs from user
	printf("enter the number of inputs you are going to provide for the operations sum and average");
	scanf("%d",&n);

	int p=n; // Store original count for average calculation

	// Loop runs until valid numbers are entered n times
	while(n>0){

		// Asking for each number input
		printf("enter the %d number ",p-n+1);
		scanf("%d",&x);

		// If input is negative, skip without decrementing n
		if(x<0){
			printf("invalid input not considered enter positive numbers only\n");
			continue;
		}

		// Adding valid number to sum
		s=s+x;

		// Decrement count only for valid input
		n--;
	}

	// Printing sum and integer average
	printf("the sum  and average of given %d valid inputs is %d and %d ",p,s,s/p);

}

or not, if present, then count the number of times it occurs in the number? Let say 12345 number,

	// Taking number input
	printf("enter the number  ");

	// Taking digit to search

	// Extracting each digit and comparing with required digit
	while(n){

		// If digit matches, increase frequency count
		if(d==n%10)
			count++;

		// Removing last digit
		n=n/10;
	}

	// If count is 0, digit not found
	if(count==0)
		printf("sorry the digit is not found\n");

	// Else digit found with repetition count
	else
		printf("yes the digit is found and it is repeated for %ld times in number ",count);

//Program to Count the Set Bits (1’s) in an Integer
void count_set(){

	int n,count=0,x;

	// Taking number input
	printf("enter the number");
	scanf("%d",&n);

	// Loop runs until all bits are processed
	while(n){

		// Extracting last bit using AND with 1
		x=n&1;

		// Adding extracted bit to count
		count=count+x;

		// Right shifting number to check next bit
		n=n>>1;
	}

	// Printing total number of set bits
	printf("the count of set bits in given integer is %d ",count);
}

//Program to Count the reSet Bits (0’s) in an Integer
void count_reset(){

	int n,count=0,x;

	// Total number of bits in integer (usually 32 bits)
	int p=sizeof(n)*8;

	// Taking number input
	printf("enter the number");
	scanf("%d",&n);

	// Loop runs for all bit positions
	while(p){

		// Extracting last bit
		x=n&1;

		// If bit is 0, increase reset count
		if(!x)
			count=count+1;

		// Decrement bit counter
		p--;

		// Right shift to process next bit
		n=n>>1;
	}

	// Printing total reset bits count
	printf("the count of reset(0) bits in given integer is %d ",count);

}
